## Task 3: Implement Ctrl+click clipboard copy/paste

**Commit:** `feat(09): add Ctrl+click clipboard copy/paste with toast indicator`

**File:** `frontend/src/features/schedule/hooks/useClipboard.ts` (NEW)

Create a custom hook for clipboard functionality:

```typescript
import { useState, useCallback } from 'react'
import { toast } from 'sonner'
import type { ScheduleAssignment, UpsertAssignmentRequest } from '../types'

interface ClipboardState {
  assignment: Pick<ScheduleAssignment, 'projectName' | 'color' | 'status'> | null
}

export function useScheduleClipboard() {
  const [clipboard, setClipboard] = useState<ClipboardState>({ assignment: null })

  const copy = useCallback((assignment: ScheduleAssignment) => {
    setClipboard({
      assignment: {
        projectName: assignment.projectName,
        color: assignment.color,
        status: assignment.status,
      },
    })
    toast.info(`Copied: ${assignment.projectName}`, { duration: 1500 })
  }, [])

  const paste = useCallback((
    personId: string,
    weekStart: string,
    year: number,
    splitIndex: number,
  ): UpsertAssignmentRequest | null => {
    if (!clipboard.assignment) {
      toast.warning('Nothing to paste')
      return null
    }
    toast.success(`Pasted: ${clipboard.assignment.projectName}`, { duration: 1500 })
    return {
      personId,
      weekStart,
      year,
      projectName: clipboard.assignment.projectName,
      color: clipboard.assignment.color,
      status: clipboard.assignment.status,
      isLocked: false,
      splitIndex,
    }
  }, [clipboard])

  const hasClipboard = clipboard.assignment !== null

  return { clipboard, copy, paste, hasClipboard }
}
```

**Wire into Schedule.tsx:**
- Call `useScheduleClipboard()` in the Schedule page
- Pass `onCellClick` handler: if Ctrl key is held AND clipboard has content Ôćĺ paste. If Ctrl key is held AND cell has assignment Ôćĺ copy. Otherwise Ôćĺ open edit modal.
- Update `onCellClick` signature or add `onCellCtrlClick` as a separate handler
- Modify `AssignmentCell` to detect Ctrl+click:
  ```typescript
  onClick={(e) => {
    if (e.ctrlKey || e.metaKey) {
      onCtrlClick(splitIndex)
    } else {
      onCellClick(splitIndex)
    }
  }}
  ```
