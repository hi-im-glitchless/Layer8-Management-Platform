## Task 2: Create API client for schedule endpoints

**Commit:** `feat(09): add schedule API client`

**File:** `frontend/src/features/schedule/api.ts` (NEW)

Follow the pattern from `features/admin/api.ts`:

```typescript
import { apiClient } from '@/lib/api'
import type {
  SchedulePerson,
  ScheduleAssignment,
  Holiday,
  Absence,
  UpsertAssignmentRequest,
  SwapAssignmentsRequest,
  SetHolidaysRequest,
  ToggleAbsenceRequest,
} from './types'

export const scheduleApi = {
  // --- Persons ---
  async getPersons() {
    return apiClient<{ persons: SchedulePerson[] }>('/api/schedule/persons')
  },

  async createPerson(name: string) {
    return apiClient<SchedulePerson>('/api/schedule/persons', {
      method: 'POST',
      body: JSON.stringify({ name }),
    })
  },

  async updatePerson(id: string, data: Partial<Pick<SchedulePerson, 'name' | 'sortOrder' | 'isActive'>>) {
    return apiClient<SchedulePerson>(`/api/schedule/persons/${id}`, {
      method: 'PUT',
      body: JSON.stringify(data),
    })
  },

  async deletePerson(id: string) {
    return apiClient<{ success: boolean }>(`/api/schedule/persons/${id}`, {
      method: 'DELETE',
    })
  },

  async reorderPersons(orderedIds: string[]) {
    return apiClient<{ success: boolean }>('/api/schedule/persons/reorder', {
      method: 'PUT',
      body: JSON.stringify({ orderedIds }),
    })
  },

  // --- Assignments ---
  async getAssignments(year: number) {
    return apiClient<{ assignments: ScheduleAssignment[] }>(`/api/schedule/assignments?year=${year}`)
  },

  async upsertAssignment(data: UpsertAssignmentRequest) {
    return apiClient<ScheduleAssignment>('/api/schedule/assignments', {
      method: 'PUT',
      body: JSON.stringify(data),
    })
  },

  async deleteAssignment(id: string) {
    return apiClient<{ success: boolean }>(`/api/schedule/assignments/${id}`, {
      method: 'DELETE',
    })
  },

  async swapAssignments(data: SwapAssignmentsRequest) {
    return apiClient<{ success: boolean }>('/api/schedule/assignments/swap', {
      method: 'POST',
      body: JSON.stringify(data),
    })
  },

  // --- Holidays ---
  async getHolidays(year: number) {
    return apiClient<{ holidays: Holiday[] }>(`/api/schedule/holidays?year=${year}`)
  },

  async setHolidays(data: SetHolidaysRequest) {
    return apiClient<{ success: boolean }>('/api/schedule/holidays', {
      method: 'PUT',
      body: JSON.stringify(data),
    })
  },

  // --- Absences ---
  async getAbsences(year: number, personId?: string) {
    const params = new URLSearchParams({ year: String(year) })
    if (personId) params.set('personId', personId)
    return apiClient<{ absences: Absence[] }>(`/api/schedule/absences?${params}`)
  },

  async toggleAbsence(data: ToggleAbsenceRequest) {
    return apiClient<{ added: boolean }>('/api/schedule/absences/toggle', {
      method: 'POST',
      body: JSON.stringify(data),
    })
  },
}
```

## Task 3: Create TanStack Query hooks for schedule

**Commit:** `feat(09): add schedule TanStack Query hooks`

**File:** `frontend/src/features/schedule/hooks.ts` (NEW)

Follow the pattern from `features/admin/hooks.ts`:

```typescript
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
import { toast } from 'sonner'
import { scheduleApi } from './api'
import type {
  UpsertAssignmentRequest,
  SwapAssignmentsRequest,
  SetHolidaysRequest,
  ToggleAbsenceRequest,
} from './types'

// Query key factory
const scheduleKeys = {
  all: ['schedule'] as const,
  persons: () => [...scheduleKeys.all, 'persons'] as const,
  assignments: (year: number) => [...scheduleKeys.all, 'assignments', year] as const,
  holidays: (year: number) => [...scheduleKeys.all, 'holidays', year] as const,
  absences: (year: number, personId?: string) => [...scheduleKeys.all, 'absences', year, personId] as const,
}

// --- Persons ---
export function usePersons() {
  return useQuery({
    queryKey: scheduleKeys.persons(),
    queryFn: () => scheduleApi.getPersons(),
  })
}

export function useCreatePerson() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (name: string) => scheduleApi.createPerson(name),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: scheduleKeys.persons() })
      toast.success('Team member added')
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to add team member'),
  })
}

export function useUpdatePerson() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: ({ id, data }: { id: string; data: Partial<{ name: string; sortOrder: number; isActive: boolean }> }) =>
      scheduleApi.updatePerson(id, data),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: scheduleKeys.persons() })
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to update team member'),
  })
}

export function useDeletePerson() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => scheduleApi.deletePerson(id),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: scheduleKeys.persons() })
      toast.success('Team member removed')
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to remove team member'),
  })
}

export function useReorderPersons() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (orderedIds: string[]) => scheduleApi.reorderPersons(orderedIds),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: scheduleKeys.persons() })
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to reorder team'),
  })
}

// --- Assignments ---
export function useAssignments(year: number) {
  return useQuery({
    queryKey: scheduleKeys.assignments(year),
    queryFn: () => scheduleApi.getAssignments(year),
  })
}

export function useUpsertAssignment() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (data: UpsertAssignmentRequest) => scheduleApi.upsertAssignment(data),
    onSuccess: (_data, vars) => {
      qc.invalidateQueries({ queryKey: scheduleKeys.assignments(vars.year) })
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to save assignment'),
  })
}

export function useDeleteAssignment(year: number) {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => scheduleApi.deleteAssignment(id),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: scheduleKeys.assignments(year) })
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to delete assignment'),
  })
}

export function useSwapAssignments(year: number) {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (data: SwapAssignmentsRequest) => scheduleApi.swapAssignments(data),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: scheduleKeys.assignments(year) })
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to swap assignments'),
  })
}

// --- Holidays ---
export function useHolidays(year: number) {
  return useQuery({
    queryKey: scheduleKeys.holidays(year),
    queryFn: () => scheduleApi.getHolidays(year),
  })
}

export function useSetHolidays() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (data: SetHolidaysRequest) => scheduleApi.setHolidays(data),
    onSuccess: (_data, vars) => {
      qc.invalidateQueries({ queryKey: scheduleKeys.holidays(vars.year) })
      toast.success('Holidays updated')
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to update holidays'),
  })
}

// --- Absences ---
export function useAbsences(year: number, personId?: string) {
  return useQuery({
    queryKey: scheduleKeys.absences(year, personId),
    queryFn: () => scheduleApi.getAbsences(year, personId),
  })
}

export function useToggleAbsence(year: number) {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (data: ToggleAbsenceRequest) => scheduleApi.toggleAbsence(data),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: scheduleKeys.absences(year) })
      qc.invalidateQueries({ queryKey: scheduleKeys.assignments(year) })
    },
    onError: (e: Error) => toast.error(e.message || 'Failed to toggle absence'),
  })
}
```

## Task 4: Create Schedule page route (placeholder shell)

**Commit:** `feat(09): add Schedule page route shell`

**File:** `frontend/src/routes/Schedule.tsx` (NEW)

Create a minimal page shell that loads the schedule data and renders a placeholder. This will be fleshed out in Plan 03.

```typescript
import { useState } from 'react'
import { CalendarDays } from 'lucide-react'
import { usePersons, useAssignments, useHolidays, useAbsences } from '@/features/schedule/hooks'

export function Schedule() {
  const currentYear = new Date().getFullYear()
  const [year] = useState(currentYear)

  const personsQuery = usePersons()
  const assignmentsQuery = useAssignments(year)
  const holidaysQuery = useHolidays(year)
  const absencesQuery = useAbsences(year)

  const isLoading = personsQuery.isLoading || assignmentsQuery.isLoading ||
    holidaysQuery.isLoading || absencesQuery.isLoading

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold tracking-tight flex items-center gap-3">
            <CalendarDays className="h-8 w-8" />
            Team Schedule
          </h1>
          <p className="text-muted-foreground mt-2">
            {year} team allocation and availability
          </p>
        </div>
      </div>

      {isLoading ? (
        <div className="flex items-center justify-center py-20">
          <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-primary" />
        </div>
      ) : (
        <div className="text-muted-foreground text-center py-20">
          Calendar grid will be rendered here
        </div>
      )}
    </div>
  )
}
```
