# Plan 03: Calendar Grid UI with Quarterly Tabs and Sticky Headers

## Overview

Build the core visual grid: a yearly calendar with team members as rows and weeks as columns, organized into quarterly sections with an "All" tab. Uses sticky positioning for scrollable navigation and renders color-coded assignment cells with availability dots.

## Task 4: Wire ScheduleGrid into Schedule page

**Commit:** `feat(09): wire ScheduleGrid into Schedule page with data loading`

**File:** `frontend/src/routes/Schedule.tsx` (MODIFY — replace the placeholder content from Plan 02)

Update the Schedule page to:
1. Import and render `ScheduleGrid`
2. Pass loaded data from hooks as props
3. Derive `canEdit` from `useAuth()` + `hasRole(role, 'MANAGER')`
4. Handle `onCellClick` by storing `{ personId, weekStart, splitIndex }` in state (modal trigger — actual modal comes in Plan 04)
5. Handle `onDotClick` by calling `useToggleAbsence` mutation

```typescript
import { useState, useMemo } from 'react'
import { CalendarDays } from 'lucide-react'
import { usePersons, useAssignments, useHolidays, useAbsences, useToggleAbsence } from '@/features/schedule/hooks'
import { useAuth } from '@/features/auth/hooks'
import { hasRole } from '@/lib/rbac'
import { ScheduleGrid } from '@/features/schedule/components/ScheduleGrid'

export function Schedule() {
  const currentYear = new Date().getFullYear()
  const [year] = useState(currentYear)
  const { role } = useAuth()
  const canEdit = hasRole(role, 'MANAGER')

  const personsQuery = usePersons()
  const assignmentsQuery = useAssignments(year)
  const holidaysQuery = useHolidays(year)
  const absencesQuery = useAbsences(year)
  const toggleAbsence = useToggleAbsence(year)

  const persons = personsQuery.data?.persons ?? []
  const assignments = assignmentsQuery.data?.assignments ?? []
  const holidays = holidaysQuery.data?.holidays ?? []
  const absences = absencesQuery.data?.absences ?? []

  const isLoading = personsQuery.isLoading || assignmentsQuery.isLoading ||
    holidaysQuery.isLoading || absencesQuery.isLoading

  // Cell click state (for edit modal — wired in Plan 04)
  const [editTarget, setEditTarget] = useState<{ personId: string; weekStart: string; splitIndex: number } | null>(null)

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold tracking-tight flex items-center gap-3">
            <CalendarDays className="h-8 w-8" />
            Team Schedule
          </h1>
          <p className="text-muted-foreground mt-1">{year} team allocation and availability</p>
        </div>
        {/* Team panel + Holiday config buttons go here (Plan 04) */}
      </div>

      {isLoading ? (
        <div className="flex items-center justify-center py-20">
          <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-primary" />
        </div>
      ) : (
        <ScheduleGrid
          year={year}
          persons={persons}
          assignments={assignments}
          holidays={holidays}
          absences={absences}
          canEdit={canEdit}
          onCellClick={(personId, weekStart, splitIndex) =>
            canEdit && setEditTarget({ personId, weekStart, splitIndex })
          }
          onDotClick={(personId, date) =>
            canEdit && toggleAbsence.mutate({ personId, date })
          }
        />
      )}
    </div>
  )
}
```
